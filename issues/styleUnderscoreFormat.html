# 下划线转换驼峰

简单对象（一维）实现转换比较容易，可以使用 lodash 的 mapKeys方法配合正则轻松完成

/**
 * @param { Object } data
 * @returns { Object }
 */

const _ = require('./lodash')
const result = _.mapKeys(data, (value, key) => key.replace(/_(\w)/g, (arg0, arg1) => arg1.toUpperCase()))

深度处理复杂数据类型( Object、Array )的key下划线 转换 驼峰命名

  • 封装方法
/**
 * @param { Object } data
 * @returns { Object }
 */

const getCamelCase = data => {
  const targetData = data.constructor === Object ? {} : []
  const hasUnderscore = /_(\w)/g
  const styleUnderscoreFormat = (arg0, arg1) => arg1.toUpperCase()

  Object.keys(data).forEach(keys => {
    if (data[keys] && typeof data[keys] === 'object') {
      targetData[keys.replace(hasUnderscore, styleUnderscoreFormat)] = getCamelCase(data[keys])
    } else {
      targetData[keys.replace(hasUnderscore, styleUnderscoreFormat)] = data[keys]
    }
  })
  return targetData
}
